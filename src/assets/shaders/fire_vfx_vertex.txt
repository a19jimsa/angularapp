#version 300 es
layout(location = 0) in vec3 a_position;
layout(location = 1) in vec2 a_texcoord;

layout(location = 2) in vec3 a_instances;
layout(location = 3) in float a_age;
layout(location = 4) in float a_lifetime;
layout(location = 5) in vec3 a_color;
layout(location = 6) in float a_size;
layout(location = 7) in float a_angle;

uniform mat4 u_matrix;
uniform mat4 u_model;

out vec2 v_texcoord;
out float v_lifetime;
out float v_age;
out vec3 v_color;
out float v_size;

vec3 rotateZ(float angle, vec3 pos){
    float c = cos(angle);
    float s = sin(angle);

    // Rotera kring Y-axeln (normals riktning ändras också)
    
    return vec3(
        pos.x * c - pos.y * s,
        pos.x * s + pos.y * c,
        pos.z
    );
}
 
void main() {
    v_texcoord = a_texcoord;
    v_age = a_age;
    v_lifetime = a_lifetime;
    v_color = a_color;
    v_size = a_size;
    vec3 pos = a_position * a_size;
    vec3 rotatedPos = rotateZ(a_angle, pos);
    gl_Position = u_matrix * vec4(rotatedPos + a_instances, 1.0);
}
