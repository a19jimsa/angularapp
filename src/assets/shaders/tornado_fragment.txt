#version 300 es
precision mediump float;
precision mediump sampler2DArray;

uniform sampler2DArray u_textures;
uniform sampler2D u_texture;
uniform float u_time;
uniform vec2 u_speed;

in vec2 v_texcoord;
in float v_lifetime;
in float v_age;
in vec3 v_color;

out vec4 FragColor;

void main() { 
    float t = 1.0 - v_age / v_lifetime;
    vec2 uv = v_texcoord;
    vec2 uvScrolling = uv + vec2(t * u_speed.x, t * u_speed.y); 
    vec3 baseColor = texture(u_texture, uvScrolling).rgb;
    vec3 color = mix(vec3(1.0, 1.0, 0.0), vec3(1.0, 0.0, 0.0), baseColor);
    FragColor = vec4(color, t);
}
