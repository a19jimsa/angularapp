#version 300 es
precision mediump float;
precision mediump sampler2DArray;

uniform sampler2DArray u_textures;
uniform sampler2D u_texture;
uniform sampler2D u_texture2;
uniform float u_time;
uniform vec2 u_speed;

in vec2 v_texcoord;
in float v_lifetime;
in float v_age;
in vec3 v_color;

out vec4 FragColor;

void main() { 
    float t = 1.0 - v_age / v_lifetime;
    vec2 uv = v_texcoord;
    vec2 uvScrolling = uv + vec2(u_time * u_speed.x, u_time * u_speed.y); 
    vec4 baseColor = texture(u_texture, uvScrolling);
    vec4 erase = texture(u_texture2, uv);
    vec3 color = mix(vec3(0.0, 0.5, 0.0), vec3(1.0, 1.0, 1.0), baseColor.r);
    color *= baseColor.r * erase.r;
    FragColor = vec4(color, t);
}
