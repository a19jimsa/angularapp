#version 300 es
precision mediump float;
precision mediump sampler2DArray;

uniform sampler2D u_texture;
uniform sampler2DArray u_textures;
uniform float u_time;
uniform vec2 u_speed;

in vec2 v_texcoord;
in float v_lifetime;
in float v_age;
in vec3 v_color;

out vec4 FragColor;

float remap(float value, float inMin, float inMax, float outMin, float outMax)
{
    return outMin + (value - inMin) * (outMax - outMin) / (inMax - inMin);
}

void main() { 
    float t = clamp(v_age / v_lifetime, 0.0, 1.0);
    vec2 uv = v_texcoord + u_speed * u_time;
    vec4 baseColor = texture(u_texture, uv);
    vec4 colorRamp = texture(u_textures, vec3(vec2(t, 0.5), 0));
    if(baseColor.r <= 0.05)
        discard;

    float x = remap(colorRamp.r, 0.0, 1.0, 0.0, 0.7);

    float opacity = smoothstep(x, 0.7, baseColor.r);

    vec3 color = texture(u_textures, vec3(vec2(baseColor.r, 0.5), 1)).rgb;

    FragColor = vec4(color, opacity);
}
