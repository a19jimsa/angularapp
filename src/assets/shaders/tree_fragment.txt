#version 300 es
precision mediump float;
precision mediump sampler2DArray;

//Shared in shaders
uniform sampler2Darray u_textures;
uniform float u_time;

in vec2 v_texcoord;
in float v_fogDistance;
in float v_index;

out vec4 FragColor;

void main() {
    vec4 texColor = texture(u_textures, vec3(v_texcoord, 0));
    if(texColor.a < 1.0)
        discard;
    float fogFactor = 1.0 - exp(-v_fogDistance * 0.0005);
    vec3 fogColor = vec3(0.5, 0.6, 0.75);
    float alpha = 1.0;
    vec3 color = mix(texColor.rgb, fogColor, fogFactor);
    FragColor = vec4(color, alpha);
}
