#version 300 es
precision mediump float;

in vec2 v_texcoord;
in float v_height;

in vec3 v_normal;
in float v_depth;
in float v_fogDistance;
in vec4 v_worldSpace;
in vec3 v_cameraPos;
in vec3 v_fragPos;

out vec4 FragColor;

uniform sampler2D u_texture;
uniform sampler2D u_splatmap;
uniform sampler2D u_brushTexture;

uniform float u_fogPower;
uniform float u_tiling;
uniform vec3 u_cameraPos;
uniform vec2 u_tiles[4];
uniform vec3 u_fogColor;
uniform vec2 u_brushUV;

struct Material{
   vec3 ambient;
   vec3 diffuse;
   vec3 specular;
   float shininess;
};

uniform Material material;

struct Light{
   vec3 position;

   vec3 ambient;
   vec3 diffuse;
   vec3 specular;
};

uniform Light light;


vec4 TriplarMapping(vec3 worldNormal, sampler2D bajs, vec2 tile){
   vec2 pixelResolution = vec2(6144, 2048);
   vec2 pixelOffset = 1.0 / pixelResolution;
   vec2 padding = pixelOffset * 0.5;
   vec2 tileSize = vec2(1.0 / 6.0, 1.0 / 2.0);
   float tiling = u_tiling;

   vec3 blending = abs(worldNormal);

   blending = pow(blending, vec3(32.0));

   blending /= (blending.x + blending.y + blending.z);

   vec2 uvX = v_worldSpace.zy * tiling;
   vec2 uvY = v_worldSpace.xz * tiling;
   vec2 uvZ = -v_worldSpace.xy * tiling;

   vec4 colorX = texture(bajs, fract(uvX) * (tileSize - 2.0 * padding) + tile * tileSize + padding);
   vec4 colorY = texture(bajs, fract(uvY) * (tileSize - 2.0 * padding) + tile * tileSize + padding);
   vec4 colorZ = texture(bajs, fract(uvZ) * (tileSize - 2.0 * padding) + tile * tileSize + padding);

   vec4 finalColor = colorX * blending.x + colorY * blending.y + colorZ * blending.z;
   return finalColor;
}

void main() {
   vec3 norm = normalize(v_normal);
   vec4 leaf_color = TriplarMapping(norm, u_texture, u_tiles[0]);
   vec4 sand_color = TriplarMapping(norm, u_texture, u_tiles[1]);
   vec4 snow_color = TriplarMapping(norm, u_texture, u_tiles[2]);
   vec4 ground_color = TriplarMapping(norm, u_texture, u_tiles[3]);

   vec4 splat = texture(u_splatmap, v_texcoord);

   vec2 position = u_brushUV;
   vec2 size = vec2(0.2); // hur stor brushen ska vara
   vec2 brushUV = (v_texcoord - position) / size + 0.5;
   vec4 brush = texture(u_brushTexture, brushUV);

   vec4 splatColor =
      leaf_color * splat.r +
      sand_color  * splat.g +
      snow_color  * splat.b +
      ground_color * splat.a;

   // ambient
   vec3 ambient = light.ambient * splatColor.rgb;

   // diffuse 
   // vec3 lightDir = normalize(light.position - v_fragPos);
   vec3 lightDir = normalize(-vec3(-0.5f, -0.0f, -0.2f));
   float diff = max(dot(norm, lightDir), 0.0);
   vec3 diffuse = light.diffuse * diff * splatColor.rgb;

   // specular
   vec3 viewDir = normalize(u_cameraPos - v_fragPos);
   vec3 reflectDir = reflect(-lightDir, norm);  
   float spec = pow(max(dot(viewDir, reflectDir), 0.0), material.shininess);
   vec3 specular = light.specular * spec * splatColor.rgb;

   float fogFactor = smoothstep(1500.0 , 5000.0 * u_fogPower, v_fogDistance);

   vec3 fogColor = u_fogColor;
   vec3 color = mix(splatColor.rgb, fogColor, fogFactor);

   vec3 result = (ambient + diffuse + specular) * color;

   vec3 finalColor = result;

   if(brushUV.x >= 0.0 && brushUV.x <= 1.0 &&
      brushUV.y >= 0.0 && brushUV.y <= 1.0) {

      vec4 brush = texture(u_brushTexture, brushUV);

      float highlightMask = 1.0 - brush.r;

      vec3 highlightColor = vec3(1.0, 0.5, 1.0); // orange

      finalColor = mix(finalColor, highlightColor, highlightMask * 0.5);
   }

   FragColor = vec4(result, 1.0);
}
