#version 300 es
precision mediump float;
precision mediump sampler2DArray;

in vec2 v_texcoord;
in float v_height;

in vec3 v_normal;
in float v_depth;
in float v_fogDistance;
in vec4 v_worldSpace;
in vec3 v_cameraPos;
in vec3 v_fragPos;

out vec4 FragColor;

uniform sampler2DArray u_brushes;
uniform sampler2DArray u_textures;
uniform sampler2D u_splatmap;

uniform float u_fogPower;
uniform float u_tiling;
uniform vec3 u_cameraPos;
uniform vec2 u_tiles[4];
uniform vec3 u_fogColor;
uniform vec2 u_brushUV;
uniform float u_brushLayer;
uniform float u_brushSize;

struct Material{
   vec3 ambient;
   vec3 diffuse;
   vec3 specular;
   float shininess;
};

uniform Material material;

struct Light{
   vec3 position;
   vec3 direction;
   vec3 ambient;
   vec3 diffuse;
   vec3 specular;
};

uniform Light light;


vec4 TriplarMapping(vec3 worldNormal, sampler2DArray textures, int layer){
   float tiling = u_tiling;

   vec3 blending = abs(worldNormal);

   blending = pow(blending, vec3(32.0));

   blending /= (blending.x + blending.y + blending.z);

   vec2 uvX = v_worldSpace.zy * tiling;
   vec2 uvY = v_worldSpace.xz * tiling;
   vec2 uvZ = v_worldSpace.xy * tiling;

   vec4 colorR = texture(textures, vec3(uvX, layer));
   vec4 colorG = texture(textures, vec3(uvY, layer));
   vec4 colorB = texture(textures, vec3(uvZ, layer));

   vec4 finalColor = colorR * blending.x + colorG * blending.y + colorB * blending.z;
   return finalColor;
}

void main() {
   //Normal in worldspace
   vec3 worldNormal = normalize(v_normal);

   vec4 leaf_color = TriplarMapping(worldNormal, u_textures, 0);
   vec4 sand_color = TriplarMapping(worldNormal, u_textures, 1);
   vec4 snow_color = TriplarMapping(worldNormal, u_textures, 2);
   vec4 ground_color = TriplarMapping(worldNormal, u_textures, 3);
   vec4 ground_normal = TriplarMapping(worldNormal, u_textures, 4);

   vec4 splat = texture(u_splatmap, v_texcoord);

   vec4 splatColor =
      leaf_color * splat.r +
      sand_color  * splat.g +
      snow_color  * splat.b +
      ground_color * splat.a;
   
   vec3 splatNormal =
      worldNormal.rgb * splat.r +
      worldNormal.rgb * splat.g +
      worldNormal.rgb * splat.b +
      ground_normal.rgb * splat.a;

   vec3 normal = normalize(worldNormal + splatNormal);

   // ambient
   vec3 ambient = material.ambient;

   // diffuse 
   // vec3 lightDir = normalize(light.position - v_fragPos);
   vec3 lightDir = normalize(-light.direction);
   float diff = max(dot(normal, lightDir), 0.0);
   vec3 diffuse = material.diffuse * diff;

   vec3 result = (ambient + diffuse) * splatColor.rgb;

   float fogFactor = smoothstep(0.0 , 500.0 * u_fogPower, v_fogDistance);
   vec3 fogColor = u_fogColor;
   vec3 color = mix(result, fogColor, fogFactor);

   vec3 finalColor = color;

   //Brush size and position
   vec2 position = u_brushUV;
   vec2 size = vec2(u_brushSize * 0.01); // hur stor brushen ska vara
   vec2 brushUV = (v_texcoord - position) / size + 0.5;

   if(brushUV.x >= 0.0 && brushUV.x <= 1.0 &&
      brushUV.y >= 0.0 && brushUV.y <= 1.0) {

      vec4 brush = texture(u_brushes, vec3(brushUV, u_brushLayer));

      float highlightMask = brush.r;

      vec3 highlightColor = vec3(0.0, 0.2, 1.0); // blue

      finalColor = mix(finalColor, highlightColor, highlightMask * 0.8);
   }
   FragColor = vec4(finalColor, 1.0);
}
