varying vec3 v_instancePositions;

    float scale = 0.8;
    float blend = smoothstep(0.2, 0.8, v_random_Id);
    vec3 color = mix(vec3(0.0, 0.3, 0.0), vec3(0.6, 0.9, 0.3), blend);
    vec3 finalColor = mix(vec3(0.0, 0.3, 0.0), color, v_height / 2.0); 
    //gl_FragColor = texColor;
    gl_FragColor = vec4(finalColor, 1.0);
}
