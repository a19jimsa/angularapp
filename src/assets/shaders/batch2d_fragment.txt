#version 300 es
precision mediump float;
precision mediump sampler2DArray;

uniform sampler2DArray u_textures;

in vec2 v_texcoord;
in float v_texIndex;

out vec4 FragColor;

void main() {
    vec4 color = texture(u_textures, vec3(v_texcoord, 1));
    if(color.a < 0.9)
        discard;
    FragColor = color;
}
