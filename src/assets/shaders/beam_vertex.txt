#version 300 es
layout(location = 0) in vec3 a_position;
layout(location = 1) in vec2 a_texcoord;

layout(location = 2) in vec3 a_instances;
layout(location = 3) in float a_age;
layout(location = 4) in float a_lifetime;
layout(location = 5) in vec3 a_color;
layout(location = 6) in float a_size;
layout(location = 7) in float a_rotationSpeed;
layout(location = 8) in float a_sizeBegin;
layout(location = 9) in float a_sizeEnd;
layout(location = 10) in vec3 a_rotation;

uniform mat4 u_matrix;
uniform mat4 u_model;
uniform sampler2D u_texture;

out vec2 v_texcoord;
out float v_lifetime;
out float v_age;
out vec3 v_color;

vec3 rotateX(vec3 p, float angle)
{
    float c = cos(angle);
    float s = sin(angle);

    return vec3(
        p.x,
        p.y * c - p.z * s,
        p.y * s + p.z * c
    );
}

vec3 rotateY(vec3 p, float angle)
{
    float c = cos(angle);
    float s = sin(angle);

    return vec3(
        p.x * c + p.z * s,
        p.y,
        -p.x * s + p.z * c
    );
}

vec3 rotateZ(vec3 p, float angle)
{
    float c = cos(angle);
    float s = sin(angle);

    return vec3(
        p.x * c - p.y * s,
        p.x * s + p.y * c,
        p.z
    );
}

void main() {
    v_texcoord = a_texcoord;
    v_age = a_age;
    v_lifetime = a_lifetime;
    v_color = a_color;
    vec3 pos = a_position * a_size;
    float t = v_age / v_lifetime;
    float size = mix(a_sizeBegin, a_sizeEnd, t);
    pos *= size;
    pos = rotateX(pos, a_rotation.x);
    pos = rotateY(pos, a_rotation.y * t);
    pos = rotateZ(pos, a_rotation.z);
    gl_Position = u_matrix * vec4(pos + a_instances, 1.0);
}
