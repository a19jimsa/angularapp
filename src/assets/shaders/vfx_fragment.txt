#version 300 es
precision mediump float;
precision mediump sampler2DArray;

uniform sampler2DArray u_textures;
uniform float u_time;
uniform vec2 u_speed;

in vec2 v_texcoord;
in float v_lifetime;
in float v_age;
in vec3 v_color;

out vec4 FragColor;

void main() { 
    float t = 1.0 - v_age / v_lifetime;
    vec2 uv = v_texcoord;
    vec4 baseColor = texture(u_textures, vec3(v_texcoord, 0));
    float curveColor = texture(u_textures, vec3(t, 0.5, 1)).r;
    float opacity = baseColor.r * curveColor;
    FragColor = vec4(1.0);
}
